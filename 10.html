	function sort(arr){
		if(arr.length<=0) return []; 
		let leftArr = [];
		let rightArr = [];

		let middle = arr[0];

		for(var i=1;imiddle){
				rightArr.push(arr[i]);
			}else{
				leftArr.push(arr[i]);
			}
		}

		return [].concat(sort(leftArr),[middle],sort(rightArr));
	}
