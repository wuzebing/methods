函数柯里化
add(1)(22)(3)(41)(5)() 输出全部数相加
arguments.callee
arr.reduce(function(total,num){return total+num;},0)
	window.onload = function(){

		function add(){
			var arr = [arguments[0]];

			var _add = function(val){
				if(val){
					arr.push(val);
					return arguments.callee;//获取到当前的函数体 _add
				}else{
					return arr.reduce(function(total,num){return total+num;},0)//最后相加
				}
			}
			return _add;
		}

		var result = add(1)(22)(3)(41)(5)();
		console.log(result);
	}
